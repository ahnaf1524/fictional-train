#include <cstdio>
int main() {
  int n, m, cnt = 0;
  scanf("%d %d", &n, &m);
  for (int i = 1; i <= n; ++i) {
    if (i % 2 == 0) {
      ++cnt;
      if (cnt % 2 == 0) {
        printf("#");
        for (int j = 1; j <= m - 1; ++j) {
          printf(".");
        }
        printf("\n");
      } else {
        for (int j = 1; j <= m - 1; ++j) {
          printf(".");
        }
        printf("#");
        printf("\n");
      }
    } else {
      for (int j = 1; j <= m; ++j) {
        printf("#");
      }
      printf("\n");
    }
  }
  return 0;
}
